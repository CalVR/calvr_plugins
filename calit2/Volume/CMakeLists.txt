SET(LIB_NAME Volume)
SET(PLUGIN_HEADERS
    Volume.h
    cvrImageSequence.h
    config.h
    vtkActorToOSG.h
)

ADD_LIBRARY(${LIB_NAME}
    "SHARED"
    ${PLUGIN_HEADERS}
    Volume.cpp
    cvrImageSequence.cpp
    vtkActorToOSG.cpp
)

set(CMAKE_CXX_FLAGS "-Wno-deprecated")

FIND_PACKAGE(Virvo REQUIRED)
FIND_PACKAGE(VTK)
FIND_PACKAGE(ITK)

CONFIGURE_FILE(config.h.in config.h)

INCLUDE_DIRECTORIES(${OSG_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(${LIB_NAME} ${OSG_LIBRARIES})

INCLUDE_DIRECTORIES(${VIRVO_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(${LIB_NAME} ${VIRVO_LIBRARIES})

IF (${VTK_FOUND})
    INCLUDE_DIRECTORIES(${VTK_INCLUDE_DIRS})
    TARGET_LINK_LIBRARIES(${LIB_NAME} ${VTK_LIBRARIES})
ENDIF (${VTK_FOUND})

IF (${ITK_FOUND})
    INCLUDE_DIRECTORIES(${ITK_INCLUDE_DIRS})
    TARGET_LINK_LIBRARIES(${LIB_NAME} ${ITK_LIBRARIES})
ENDIF (${ITK_FOUND})

ADD_CALVR_LIBRARIES(${LIB_NAME})

INSTALL(TARGETS ${LIB_NAME} DESTINATION lib/plugins)
