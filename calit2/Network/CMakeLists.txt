SET(LIB_NAME Network)

SET(PLUGIN_HEADERS
    Network.h
    NetworkWallSceneObject.h
)

ADD_LIBRARY(${LIB_NAME}
    "SHARED"
    ${PLUGIN_HEADERS}
    Network.cpp
    NetworkWallSceneObject.cpp
)

FIND_PACKAGE(JsonCpp REQUIRED)

INCLUDE_DIRECTORIES(${JsonCpp_INCLUDE_DIRS})
INCLUDE_DIRECTORIES(${OSG_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(${LIB_NAME} ${OSG_LIBRARIES})
TARGET_LINK_LIBRARIES(${LIB_NAME} ${JsonCpp_LIBRARIES})

ADD_CALVR_LIBRARIES(${LIB_NAME})


INSTALL(TARGETS ${LIB_NAME} DESTINATION lib/plugins)

OPTION(PLUGIN_NETWORK_LAYOUT "Build Layout application" OFF)

IF(PLUGIN_NETWORK_LAYOUT)
    ADD_SUBDIRECTORY(Layout)
ENDIF(PLUGIN_NETWORK_LAYOUT)
